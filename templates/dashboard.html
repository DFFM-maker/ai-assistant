// Versioning functions let currentProjects = []; async function loadProjects() { try { const response = await fetch('/api/projects'); const data = await response.json(); currentProjects = data.projects || []; const projectsList = document.getElementById('projects-list'); const projectSelect = document.getElementById('project-select'); const historySelect = document.getElementById('history-project-select'); projectsList.innerHTML = ''; projectSelect.innerHTML = ''; historySelect.innerHTML = ''; currentProjects.forEach(project => { // Lista progetti const div = document.createElement('div'); div.style.cssText = 'margin: 5px 0; padding: 8px; background: var(--bg-secondary); border-radius: 4px; font-size: 0.9em;'; div.innerHTML = ` ${project.name} v${project.current_version}
${project.description} `; projectsList.appendChild(div); // Selectors const option1 = document.createElement('option'); option1.value = project.name; option1.textContent = `${project.name} (v${project.current_version})`; projectSelect.appendChild(option1); const option2 = document.createElement('option'); option2.value = project.name; option2.textContent = `${project.name} (v${project.current_version})`; historySelect.appendChild(option2); }); } catch (e) { document.getElementById('projects-list').textContent = 'Errore caricamento progetti'; } } async function createProject() { const name = document.getElementById('new-project-name').value; const description = document.getElementById('new-project-desc').value; if (!name) { alert('Inserisci nome progetto'); return; } try { const response = await fetch('/api/projects', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({name, description}) }); const result = await response.json(); if (result.success) { document.getElementById('new-project-name').value = ''; document.getElementById('new-project-desc').value = ''; loadProjects(); } else { alert(result.error); } } catch (e) { alert('Errore creazione progetto'); } } async function saveDocument() { const projectName = document.getElementById('project-select').value; const content = document.getElementById('document-content').value; const message = document.getElementById('commit-message').value; if (!projectName || !content) { alert('Seleziona progetto e inserisci contenuto'); return; } const files = { 'document.md': content }; try { const response = await fetch(`/api/projects/${projectName}/save`, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ files, message: message || 'Documento salvato da AI Assistant' }) }); const result = await response.json(); if (result.success) { alert(`Documento salvato come ${result.version}`); document.getElementById('document-content').value = ''; document.getElementById('commit-message').value = ''; loadProjects(); } else { alert(result.error); } } catch (e) { alert('Errore salvataggio documento'); } } async function loadVersionHistory() { const projectName = document.getElementById('history-project-select').value; if (!projectName) { document.getElementById('version-history').textContent = ''; return; } try { const response = await fetch(`/api/projects/${projectName}/versions`); const data = await response.json(); const historyDiv = document.getElementById('version-history'); const compareA = document.getElementById('compare-version-a'); const compareB = document.getElementById('compare-version-b'); compareA.innerHTML = ''; compareB.innerHTML = ''; if (data.versions && data.versions.length > 0) { let historyText = `Cronologia versioni (${data.versions.length}):\n\n`; data.versions.reverse().forEach(version => { historyText += `• ${version.version} - ${version.timestamp}\n`; historyText += ` ${version.commit_message}\n`; historyText += ` Hash: ${version.hash} | Files: ${version.files.length}\n\n`; // Populate compare selectors const optionA = document.createElement('option'); optionA.value = version.version; optionA.textContent = version.version; compareA.appendChild(optionA); const optionB = document.createElement('option'); optionB.value = version.version; optionB.textContent = version.version; compareB.appendChild(optionB); }); historyDiv.textContent = historyText; } else { historyDiv.textContent = 'Nessuna versione trovata'; } } catch (e) { document.getElementById('version-history').textContent = 'Errore caricamento cronologia'; } } async function compareVersions() { const projectName = document.getElementById('history-project-select').value; const versionA = document.getElementById('compare-version-a').value; const versionB = document.getElementById('compare-version-b').value; if (!projectName || !versionA || !versionB) { alert('Seleziona progetto e due versioni da confrontare'); return; } try { const response = await fetch(`/api/projects/${projectName}/diff/${versionA}/${versionB}`); const diff = await response.json(); const diffDiv = document.getElementById('version-diff'); if (diff.error) { diffDiv.textContent = diff.error; return; } let diffText = `Confronto ${versionA} → ${versionB}:\n\n`; if (diff.added.length > 0) { diffText += `✅ Files aggiunti (${diff.added.length}): ${diff.added.join(', ')}\n`; } if (diff.modified.length > 0) { diffText += `📝 Files modificati (${diff.modified.length}): ${diff.modified.join(', ')}\n`; Object.entries(diff.changes).forEach(([file, changes]) => { diffText += ` - ${file}: ${changes.lines_added > 0 ? '+' : ''}${changes.lines_added} righe, ${changes.chars_changed} caratteri modificati\n`; }); } if (diff.deleted.length > 0) { diffText += `❌ Files eliminati (${diff.deleted.length}): ${diff.deleted.join(', ')}\n`; } if (diff.added.length === 0 && diff.modified.length === 0 && diff.deleted.length === 0) { diffText += 'Nessuna differenza trovata'; } diffDiv.textContent = diffText; } catch (e) { document.getElementById('version-diff').textContent = 'Errore confronto versioni'; } } // Load projects on page load document.addEventListener('DOMContentLoaded', function() { loadProjects(); });
